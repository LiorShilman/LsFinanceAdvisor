  <script>

    function getEl(id) 
    {
        return document.getElementById(id);
    }





    // Chart Management
    // דוגמה למחלקה – הוסף/מזג למחלקה הקיימת שלך
    class ChartManager {
      constructor() {
        this.charts = {
          pathChart: null,
          compositionChart: null,
          sensitivityChart: null
        };
      }

     
      /**
       * קו התקדמות/מסלולי הון לאורך זמן
       * results:
       *  - pathSeries: [{label, data: [numbers], labels?: [labels]}]  // עדיפות ראשונה
       *  - או yearlyBalances: [ { year, value } ]
       *  - או monthlyPath: [numbers]
       */
      createPathChart(results) {
        const canvas = getEl('pathChart');
        if (!canvas || !results) return;

        // נקה כל גרף קיים על הקנבס (גם אם נוצר ממופע אחר)
        const existing = Chart.getChart(canvas);
        if (existing) existing.destroy();

        // נקה מהאחסון הפנימי של המופע
        if (this.charts.pathChart) {
          this.charts.pathChart.destroy();
          this.charts.pathChart = null;
        }

        // --- הרכבת נתונים גמישה ---
        let labels = [];
        let datasets = [];

        if (Array.isArray(results.pathSeries) && results.pathSeries.length) {
          // תבנית כללית של כמה סדרות
          const first = results.pathSeries[0];
          labels = Array.isArray(first.labels) && first.labels.length
            ? first.labels.slice()
            : [...Array(first.data.length)].map((_, i) => (i + 1).toString());

          datasets = results.pathSeries.map((s, idx) => ({
            label: s.label || `Series ${idx + 1}`,
            data: Array.isArray(s.data) ? s.data : [],
            parsing: false,
            pointRadius: 0,
            borderWidth: 2
          }));
        } else if (Array.isArray(results.yearlyBalances) && results.yearlyBalances.length) {
          // תבנית: [{year, value}]
          labels = results.yearlyBalances.map(r => String(r.year));
          const data = results.yearlyBalances.map(r => Number(r.value || 0));
          datasets = [{
            label: 'יתרה שנתית',
            data,
            pointRadius: 0,
            borderWidth: 2
          }];
        } else if (Array.isArray(results.monthlyPath) && results.monthlyPath.length) {
          // תבנית: [numbers]
          labels = results.monthlyPath.map((_, i) => (i + 1).toString());
          datasets = [{
            label: 'מסלול חודשי',
            data: results.monthlyPath.slice(),
            pointRadius: 0,
            borderWidth: 2
          }];
        } else {
          // ברירת מחדל מינימלית – שלא יקרוס
          labels = ['1', '2', '3', '4', '5'];
          datasets = [{
            label: 'נתיב סימולציה',
            data: [0, 1, 2, 3, 4],
            pointRadius: 0,
            borderWidth: 2
          }];
        }

        // יצירת הגרף
        this.charts.pathChart = new Chart(canvas, {
          type: 'line',
          data: { labels, datasets },
          options: {
            responsive: true,
            maintainAspectRatio: false,
            interaction: { mode: 'index', intersect: false },
            plugins: {
              legend: { display: true, position: 'top' },
              title: { display: true, text: 'נתיב הון לאורך זמן' },
              tooltip: { enabled: true }
            },
            scales: {
              x: { title: { display: true, text: 'תקופות' }, ticks: { autoSkip: true, maxTicksLimit: 12 } },
              y: { title: { display: true, text: '₪' }, ticks: { callback: v => v.toLocaleString('he-IL') } }
            }
          }
        });
      }

      /**
       * הרכב/Stack (לדוגמה: תרומה של פיקדון/הפקדות/תשואה בכל תקופה)
       * results:
       *  - composition: { labels: [...], series: [{label, data:[...]}] }
       *  - או stacked: [{name, values:[...]}] + labels:[...]
       */
      createCompositionChart(results) {
        const canvas = getEl('stackChart');
        if (!canvas || !results) return;

        const existing = Chart.getChart(canvas);
        if (existing) existing.destroy();

        if (this.charts.compositionChart) {
          this.charts.compositionChart.destroy();
          this.charts.compositionChart = null;
        }

        // --- הרכבת נתונים גמישה ---
        let labels = [];
        let datasets = [];

        if (results.composition &&
          Array.isArray(results.composition.labels) &&
          Array.isArray(results.composition.series)) {
          labels = results.composition.labels.slice();
          datasets = results.composition.series.map((s, idx) => ({
            type: 'bar',
            label: s.label || `רכיב ${idx + 1}`,
            data: Array.isArray(s.data) ? s.data : [],
            stack: 'total'
          }));
        } else if (Array.isArray(results.stacked) && Array.isArray(results.labels)) {
          labels = results.labels.slice();
          datasets = results.stacked.map((s, idx) => ({
            type: 'bar',
            label: s.name || `רכיב ${idx + 1}`,
            data: Array.isArray(s.values) ? s.values : [],
            stack: 'total'
          }));
        } else {
          // ברירת מחדל
          labels = ['Y1', 'Y2', 'Y3', 'Y4'];
          datasets = [
            { type: 'bar', label: 'הפקדות', data: [10, 12, 14, 16], stack: 'total' },
            { type: 'bar', label: 'תשואה', data: [2, 4, 6, 9], stack: 'total' }
          ];
        }

        this.charts.compositionChart = new Chart(canvas, {
          data: { labels, datasets },
          options: {
            responsive: true,
            maintainAspectRatio: false,
            plugins: {
              legend: { display: true, position: 'top' },
              title: { display: true, text: 'הרכב הצטברות (Stacked)' },
              tooltip: { enabled: true }
            },
            scales: {
              x: { stacked: true, title: { display: true, text: 'תקופות' } },
              y: { stacked: true, title: { display: true, text: '₪' }, ticks: { callback: v => v.toLocaleString('he-IL') } }
            }
          }
        });
      }

      /**
       * רגישות – לדוגמה: שינוי תוצאה מול שיעור תשואה/אינפלציה/סיכון
       * results:
       *  - sensitivity: [{ x: number, y: number, label?: string }]  // scatter/line
       *  - או grid: { xs:[...], ys:[...]}  // יילקח רק ציר X בסיסי

      class MathUtils {
        static boxMuller() {
          let u = 0, v = 0;
          while (u === 0) u = Math.random();
          while (v === 0) v = Math.random();
          return Math.sqrt(-2.0 * Math.log(u)) * Math.cos(2 * Math.PI * v);
        }

        static setSeed(seed) {
          // Simple pseudo-random with seed (for reproducible results)
          Math.seedrandom = function () {
            let m = 2147483647, a = 16807, c = 12345;
            let z = seed || 1;
            return function () {
              z = (a * z + c) % m;
              return z / m;
            };
          };
          Math.random = Math.seedrandom();
        }

        static percentile(arr, p) {
          const sorted = [...arr].sort((a, b) => a - b);
          const index = (p / 100) * (sorted.length - 1);
          const lower = Math.floor(index);
          const upper = Math.ceil(index);
          const weight = index % 1;

          if (upper >= sorted.length) return sorted[sorted.length - 1];
          return sorted[lower] * (1 - weight) + sorted[upper] * weight;
        }

        static compoundReturn(rate, periods, contribution = 0) {
          if (rate === 0) return contribution * periods;
          return contribution * (Math.pow(1 + rate, periods) - 1) / rate;
        }
      }

      function calculateRequiredSavingsRate(gap, years, monthlySalary) {
        if (years <= 0 || gap <= 0) return 0;
        const annualSalary = monthlySalary * 12;
        const assumedReturn = 0.055; // 5.5% annual return
        const monthlyReturn = assumedReturn / 12;
        const periods = years * 12;

        // PMT needed to reach gap
        const requiredMonthlyPMT = gap * monthlyReturn / (Math.pow(1 + monthlyReturn, periods) - 1);
        return (requiredMonthlyPMT * 12) / annualSalary * 100;
      }

      // Monte Carlo Simulation Engine
      class SimulationEngine {
        constructor(params) {
          this.params = params;
          this.results = null;
        }




        async runSimulation() {
          const startTime = performance.now();

          // Set random seed for reproducibility
          if (this.params.randomSeed) {
            MathUtils.setSeed(this.params.randomSeed);
          }

          const {
            ageNow, ageRetire, startBalance, salary, salaryGrowth,
            empPct, erPct, voluntary, retExp, retVol, inflation,
            feeAum, feeContrib, nSims, events
          } = this.params;

          const months = Math.max(12, (ageRetire - ageNow) * 12);
          const monthlyRetExp = Math.log(1 + retExp / 100) / 12;
          const monthlyRetVol = (retVol / 100) / Math.sqrt(12);
          const monthlyInflation = Math.log(1 + inflation / 100) / 12;
          const monthlyFeeAum = feeAum / 100 / 12;

          // Map events by month
          const eventsByMonth = {};
          events.forEach(event => {
            const month = Math.round((event.age - ageNow) * 12);
            if (month >= 0 && month <= months) {
              (eventsByMonth[month] = eventsByMonth[month] || []).push(event);
            }
          });

          const paths = [];

          // Show progress
          this.showProgress(0);

          for (let sim = 0; sim < nSims; sim++) {
            const path = await this.runSinglePath({
              months, startBalance, salary, salaryGrowth,
              empPct, erPct, voluntary, monthlyRetExp, monthlyRetVol,
              monthlyFeeAum, feeContrib, eventsByMonth, ageNow
            });

            paths.push(path);

            // Update progress periodically
            if (sim % Math.max(1, Math.floor(nSims / 20)) === 0) {
              this.showProgress((sim / nSims) * 100);
              await this.sleep(1); // Allow UI update
            }
          }

          // Calculate percentiles and statistics
          this.results = this.calculateStatistics(paths, months, monthlyInflation);

          const endTime = performance.now();
          state.performance.lastRunTime = endTime - startTime;
          state.performance.totalRuns++;

          this.hideProgress();
          return this.results;
        }

        async runSinglePath(params) {
          const {
            months, startBalance, salary: initialSalary, salaryGrowth,
            empPct, erPct, voluntary, monthlyRetExp, monthlyRetVol,
            monthlyFeeAum, feeContrib, eventsByMonth, ageNow
          } = params;

          let balance = startBalance;
          let salary = initialSalary;
          let currentEmpPct = empPct;
          let currentErPct = erPct;
          let currentVoluntary = voluntary;
          let currentRetExp = monthlyRetExp;
          let currentRetVol = monthlyRetVol;
          let currentFeeAum = monthlyFeeAum;
          let sabbaticalMonths = 0;

          const balanceSeries = new Array(months + 1);
          const contributionSeries = new Array(months + 1);

          balanceSeries[0] = balance;
          contributionSeries[0] = 0;

          for (let month = 1; month <= months; month++) {
            // Apply salary growth annually
            if (month % 12 === 1 && month > 1) {
              salary *= (1 + salaryGrowth / 100);
            }

            // Apply events for this month
            if (eventsByMonth[month]) {
              eventsByMonth[month].forEach(event => {
                switch (event.type) {
                  case 'inc_contrib':
                    currentEmpPct += event.params.deltaPct || 0;
                    break;
                  case 'lump_sum':
                    balance += event.params.amount || 0;
                    break;
                  case 'inheritance':
                    balance += event.params.amount || 0;
                    break;
                  case 'switch_fund':
                    currentRetExp = Math.log(1 + (event.params.retExp || 5.5) / 100) / 12;
                    currentRetVol = ((event.params.retVol || 12) / 100) / Math.sqrt(12);
                    currentFeeAum = (event.params.feeAum || 0.8) / 100 / 12;
                    break;
                  case 'self_employed':
                    currentErPct = 0;
                    break;
                  case 'sabbatical':
                    sabbaticalMonths = Math.max(sabbaticalMonths, event.params.months || 0);
                    break;
                  case 'salary_jump':
                    salary *= (1 + (event.params.jumpPct || 0) / 100);
                    break;
                }
              });
            }

            // Calculate contributions (skip if on sabbatical)
            let monthlyContribution = 0;
            if (sabbaticalMonths > 0) {
              sabbaticalMonths--;
            } else {
              const empContrib = salary * (currentEmpPct / 100);
              const erContrib = salary * (currentErPct / 100);
              monthlyContribution = empContrib + erContrib + currentVoluntary;

              // Apply contribution fees
              if (feeContrib > 0) {
                monthlyContribution *= (1 - feeContrib / 100);
              }

              balance += monthlyContribution;
            }

            // Generate random return
            let monthlyReturn = 0;
            if (currentRetVol > 0) {
              const z = MathUtils.boxMuller();
              monthlyReturn = Math.exp((currentRetExp - 0.5 * currentRetVol * currentRetVol) + currentRetVol * z) - 1;
            } else {
              monthlyReturn = Math.exp(currentRetExp) - 1;
            }

            // Apply market shock if applicable
            if (eventsByMonth[month]) {
              eventsByMonth[month].forEach(event => {
                if (event.type === 'market_shock') {
                  monthlyReturn += (event.params.shockPct || 0) / 100;
                }
              });
            }

            // Apply return
            balance *= (1 + monthlyReturn);

            // Apply fees
            balance *= (1 - currentFeeAum);

            // Ensure non-negative balance
            balance = Math.max(0, balance);

            balanceSeries[month] = balance;
            contributionSeries[month] = (contributionSeries[month - 1] || 0) + monthlyContribution;
          }

          return { balanceSeries, contributionSeries };
        }

        calculateStatistics(paths, months, monthlyInflation) {
          // Calculate percentiles for each month
          const percentiles = { p5: [], p10: [], p25: [], p50: [], p75: [], p90: [], p95: [] };
          const contributions = { p50: [] };

          for (let month = 0; month <= months; month++) {
            const balances = paths.map(p => p.balanceSeries[month]).sort((a, b) => a - b);
            const contribs = paths.map(p => p.contributionSeries[month]).sort((a, b) => a - b);

            percentiles.p5[month] = MathUtils.percentile(balances, 5);
            percentiles.p10[month] = MathUtils.percentile(balances, 10);
            percentiles.p25[month] = MathUtils.percentile(balances, 25);
            percentiles.p50[month] = MathUtils.percentile(balances, 50);
            percentiles.p75[month] = MathUtils.percentile(balances, 75);
            percentiles.p90[month] = MathUtils.percentile(balances, 90);
            percentiles.p95[month] = MathUtils.percentile(balances, 95);

            contributions.p50[month] = MathUtils.percentile(contribs, 50);
          }

          // Calculate inflation adjustment series
          const inflationSeries = new Array(months + 1);
          inflationSeries[0] = 1;
          for (let month = 1; month <= months; month++) {
            inflationSeries[month] = inflationSeries[month - 1] * Math.exp(monthlyInflation);
          }

          return {
            paths,
            percentiles,
            contributions,
            inflationSeries,
            months,
            statistics: this.calculateAdvancedStats(paths, months)
          };
        }

        calculateAdvancedStats(paths, months) {
          const finalBalances = paths.map(p => p.balanceSeries[months]);
          const returns = paths.map(p => {
            const initial = p.balanceSeries[0];
            const final = p.balanceSeries[months];
            return initial > 0 ? (final / initial) ** (12 / months) - 1 : 0;
          });

          return {
            mean: finalBalances.reduce((a, b) => a + b) / finalBalances.length,
            median: MathUtils.percentile(finalBalances, 50),
            stdDev: this.standardDeviation(finalBalances),
            skewness: this.skewness(finalBalances),
            kurtosis: this.kurtosis(finalBalances),
            meanReturn: returns.reduce((a, b) => a + b) / returns.length,
            volatility: this.standardDeviation(returns),
            maxDrawdown: this.calculateMaxDrawdown(paths)
          };
        }

        standardDeviation(arr) {
          const mean = arr.reduce((a, b) => a + b) / arr.length;
          const variance = arr.reduce((a, b) => a + (b - mean) ** 2, 0) / arr.length;
          return Math.sqrt(variance);
        }

        skewness(arr) {
          const mean = arr.reduce((a, b) => a + b) / arr.length;
          const stdDev = this.standardDeviation(arr);
          const n = arr.length;
          const skew = arr.reduce((a, b) => a + Math.pow((b - mean) / stdDev, 3), 0) / n;
          return skew;
        }

        kurtosis(arr) {
          const mean = arr.reduce((a, b) => a + b) / arr.length;
          const stdDev = this.standardDeviation(arr);
          const n = arr.length;
          const kurt = arr.reduce((a, b) => a + Math.pow((b - mean) / stdDev, 4), 0) / n;
          return kurt - 3; // Excess kurtosis
        }

        calculateMaxDrawdown(paths) {
          let maxDrawdown = 0;
          paths.forEach(path => {
            let peak = 0;
            let maxDd = 0;

            path.balanceSeries.forEach(balance => {
              if (balance > peak) peak = balance;
              const drawdown = peak > 0 ? (peak - balance) / peak : 0;
              maxDd = Math.max(maxDd, drawdown);
            });

            maxDrawdown = Math.max(maxDrawdown, maxDd);
          });
          return maxDrawdown;
        }

        showProgress(percent) {
          const overlay = $('loadingOverlay');
          const fill = $('progressFill');
          if (overlay && fill) {
            overlay.style.display = 'flex';
            fill.style.width = `${percent}%`;
          }
        }

        hideProgress() {
          const overlay = $('loadingOverlay');
          if (overlay) {
            overlay.style.display = 'none';
          }
        }

        sleep(ms) {
          return new Promise(resolve => setTimeout(resolve, ms));
        }
      }


      // Main Simulation Runner
      async function runFullSimulation() {
        if (state.isRunning) return;

        try {
          state.isRunning = true;
          els.runBtn.disabled = true;
          els.runBtn.textContent = '🔄 מריץ סימולציה...';

          // Collect parameters
          const params = {
            ageNow: parseInt(els.ageNow.value) || 35,
            ageRetire: parseInt(els.ageRetire.value) || 67,
            startBalance: parseFloat(els.balanceNow.value) || 0,
            salary: parseFloat(els.salary.value) || 15000,
            salaryGrowth: parseFloat(els.salaryGrowthPct.value) || 2.5,
            empPct: parseFloat(els.empPct.value) || 6.0,
            erPct: parseFloat(els.erPct.value) || 6.5,
            voluntary: parseFloat(els.voluntaryMonthly.value) || 0,
            retExp: parseFloat(els.retExpPct.value) || 5.5,
            retVol: parseFloat(els.retVolPct.value) || 12.0,
            inflation: parseFloat(els.inflPct.value) || 2.5,
            feeAum: parseFloat(els.feeAumPct.value) || 0.8,
            feeContrib: parseFloat(els.feeContribPct.value) || 2.0,
            nSims: Math.max(100, Math.min(5000, parseInt(els.nSims.value) || 1000)),
            randomSeed: parseInt(els.randomSeed.value) || null,
            events: [...state.events]
          };

          // Run simulation
          const engine = new SimulationEngine(params);
          const results = await engine.runSimulation();
          state.lastResults = results;

          // Update KPIs
          const finalMonth = results.months;
          const finalNominal = results.percentiles.p50[finalMonth];
          const finalReal = finalNominal / results.inflationSeries[finalMonth];

          els.kpiNom.textContent = fmtCurrency(finalNominal);
          els.kpiReal.textContent = fmtCurrency(finalReal);

          // Calculate success probability
          const targetPension = parseFloat(els.targetPensionMonthly.value) || 12000;
          const swr = parseFloat(els.swrPct.value) / 100 || 0.035;
          const requiredCapital = (targetPension * 12) / swr;

          const successCount = results.paths.filter(p =>
            p.balanceSeries[finalMonth] >= requiredCapital
          ).length;
          const successRate = successCount / results.paths.length;

          els.kpiProb.textContent = `${fmt(successRate * 100, 1)}%`;

          // Update success indicator
          const indicator = $('successIndicator');
          if (indicator) {
            indicator.className = `status-indicator mx-auto mt-2 ${successRate >= 0.8 ? 'status-success' :
              successRate >= 0.6 ? 'status-warning' : 'status-danger'
              }`;
          }

          // Update expected pension
          const expectedPension = $('expectedPension');
          if (expectedPension) {
            const monthlyPension = (finalReal * swr) / 12;
            expectedPension.textContent = fmtCurrency(monthlyPension);
          }

          // Create charts
          chartManager.createPathChart(results);
          chartManager.createCompositionChart(results);
          chartManager.createSensitivityChart();

          // Update analysis and recommendations
          updateAnalysisAndRecommendations(results, params);

          // Update performance metrics
          $('lastRunTime').textContent = `${fmt(state.performance.lastRunTime, 0)}ms`;
          $('simulationCount').textContent = fmt(state.performance.totalRuns);
          $('lastExecutionTime').textContent = `${fmt(state.performance.lastRunTime, 0)}ms`;

        } catch (error) {
          console.error('Simulation error:', error);
          alert('שגיאה בהרצת הסימולציה. אנא נסה שנית.');
        } finally {
          state.isRunning = false;
          els.runBtn.disabled = false;
          els.runBtn.textContent = '🚀 הרץ סימולציה מלאה';
        }
      }

            type: 'danger',
            text: `הסתברות הצלחה נמוכה של ${fmt(successRate * 100, 1)}% מחייבת שינויים משמעותיים במסלול.`
          });
          recommendations.push({
            priority: 'critical',
            action: 'הגדל משמעותית את ההפקדה או דחה פרישה',
            impact: `נדרש חיסכון נוסף של ₪${fmt(gap / (yearsToRetire * 12))} לחודש`
          });
        }

        const volatility = results.statistics.volatility;

          insights.push({

            text: `ירידה מקסימלית של ${fmtPercent(maxDrawdown)} מצביעה על רמת סיכון גבוהה.`

            text: `דמי ניהול של ${fmt(params.feeAum, 2)}% גבוהים מהממוצע בשוק.`
          });

            priority: 'high',
            action: 'בדוק אפשרויות לקרנות בדמי ניהול נמוכים יותר',
            impact: `הפחתה ל-0.6% יכולה לחסוך ₪${fmt((params.feeAum - 0.6) * finalBalance / 100)} בפרישה`
          });
        }

        // Contribution rate analysis
        const totalContribRate = params.empPct + params.erPct +
          ((params.voluntary * 12) / (params.salary * 12)) * 100;

        if (totalContribRate < 15) {
          recommendations.push({
            priority: 'medium',
            action: 'שקול הגדלת שיעור החיסכון ל-15% מהשכר לפחות',

            priority: 'low',
            action: 'נצל את הזמן הארוך - שקול אלוקציה אגרסיבית יותר',
            impact: 'גיל צעיר מאפשר לקחת סיכונים גבוהים יותר לתשואה גבוהה'

          });

        updateRecommendationsDisplay(recommendations);
        updateActionTimeline(params, results);
        updateMainConclusion(results, params);
        updateCompositionalInsights(results);

      // Stress Testing
      function runStressTest(scenario) {
        const button = $(`stress${scenario}Btn`);
        const result = $(`stress${scenario}Result`);

        if (!button || !result) return;

        button.disabled = true;
        button.textContent = 'מריץ...';

        // Simulate stress test (simplified)
        setTimeout(() => {
          const baseResults = state.lastResults;
          if (!baseResults) {
            result.innerHTML = '<span class="text-red-400">הרץ תחילה סימולציה בסיסית</span>';
            return;
          }

          const baseMedian = baseResults.percentiles.p50[baseResults.months];
          let stressImpact = 0;
          let impactText = '';

          switch (scenario) {
            case '2008':
              stressImpact = -0.25; // 25% reduction
              impactText = `צבירה צפויה: ${fmtCurrency(baseMedian * (1 + stressImpact))} (-25%)`;
              break;
            case 'Inflation':
              stressImpact = -0.15; // 15% reduction in real terms
              impactText = `כח קנייה: ${fmtCurrency(baseMedian * (1 + stressImpact))} (-15% ריאלי)`;
              break;
            case 'JobLoss':
              stressImpact = -0.12; // 12% reduction
              impactText = `השפעה: ${fmtCurrency(baseMedian * (1 + stressImpact))} (-12%)`;
              break;
            case 'Emergency':
              stressImpact = -0.08; // 8% reduction
              impactText = `צבירה לאחר חירום: ${fmtCurrency(baseMedian * (1 + stressImpact))} (-8%)`;
              break;
          }

          const impactClass = Math.abs(stressImpact) > 0.2 ? 'text-red-400' :
            Math.abs(stressImpact) > 0.1 ? 'text-orange-400' : 'text-yellow-400';

          result.innerHTML = `<span class="${impactClass}">${impactText}</span>`;

          button.disabled = false;
          button.textContent = 'הרץ תרחיש';
        }, 1500);
      }

    // Advanced Mathematics Library Extension
    (function MathExtensions() {
      // Black-Scholes option pricing for advanced scenarios
      window.BlackScholes = {
        call: function (S, K, T, r, sigma) {
          const d1 = (Math.log(S / K) + (r + 0.5 * sigma * sigma) * T) / (sigma * Math.sqrt(T));
          const d2 = d1 - sigma * Math.sqrt(T);
          return S * this.normalCDF(d1) - K * Math.exp(-r * T) * this.normalCDF(d2);
        },

        normalCDF: function (x) {
          return 0.5 * (1 + this.erf(x / Math.sqrt(2)));
        },

        erf: function (x) {
          const a1 = 0.254829592;
          const a2 = -0.284496736;
          const a3 = 1.421413741;
          const a4 = -1.453152027;
          const a5 = 1.061405429;
          const p = 0.3275911;

          const sign = x >= 0 ? 1 : -1;
          x = Math.abs(x);

          const t = 1.0 / (1.0 + p * x);
          const y = 1.0 - (((((a5 * t + a4) * t) + a3) * t + a2) * t + a1) * t * Math.exp(-x * x);

          return sign * y;
        }
      };
    })();

          const avgTime = metrics.avgSimTime;

          const color = avgTime < 2000 ? 'text-green-400' :
